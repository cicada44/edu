    8.3.1 Использование класса istringstream

string line, word; // string and word from stream
vector<PersonInfo> people;
while (getline(cin, line)){
    PersonInfo info;
    istringstream record(line);
    record >> info.name;
    while (record >> word)
        info.push_back(word);
    people.push_back(info);
}
